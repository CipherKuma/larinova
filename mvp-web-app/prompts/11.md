-- Create prescription for Michael Chen's consultation
WITH new_rx AS (
  INSERT INTO larinova_prescriptions (consultation_id, patient_id, doctor_id, doctor_notes, email_sent_at)
  SELECT
    c.id,
    c.patient_id,
    c.doctor_id,

11. Verify data relationships:
```bash
supabase db remote shell -c "
SELECT
  p.full_name as patient,
  COUNT(DISTINCT hr.id) as health_records,
  COUNT(DISTINCT i.id) as insurance_policies,
  COUNT(DISTINCT c.id) as consultations,
  COUNT(DISTINCT pr.id) as prescriptions
FROM larinova_patients p
LEFT JOIN larinova_health_records hr ON p.id = hr.patient_id
LEFT JOIN larinova_insurance i ON p.id = i.patient_id
LEFT JOIN larinova_consultations c ON p.id = c.patient_id
LEFT JOIN larinova_prescriptions pr ON p.id = pr.patient_id
GROUP BY p.full_name
ORDER BY consultations DESC, health_records DESC;
"
```
