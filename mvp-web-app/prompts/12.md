9. Check for orphaned data:
```bash
supabase db remote shell -c "
SELECT
  'Orphaned Records Check' as check_name,
  '========================' as divider
UNION ALL
SELECT
  'Consultations without patients',
  COUNT(*)::text
FROM larinova_consultations c
LEFT JOIN larinova_patients p ON c.patient_id = p.id
WHERE p.id IS NULL
UNION ALL
SELECT
  'Prescriptions without consultations',
  COUNT(*)::text
FROM larinova_prescriptions pr
LEFT JOIN larinova_consultations c ON pr.consultation_id = c.id
WHERE c.id IS NULL
UNION ALL
SELECT
  'Transcripts without consultations',
  COUNT(*)::text
FROM larinova_transcripts t
LEFT JOIN larinova_consultations c ON t.consultation_id = c.id
WHERE c.id IS NULL
UNION ALL
SELECT
  'Result',
  CASE
    WHEN (
      SELECT COUNT(*) FROM larinova_consultations c
      LEFT JOIN larinova_patients p ON c.patient_id = p.id
      WHERE p.id IS NULL
    ) = 0 AND (
      SELECT COUNT(*) FROM larinova_prescriptions pr
      LEFT JOIN larinova_consultations c ON pr.consultation_id = c.id
      WHERE c.id IS NULL
    ) = 0
    THEN '✅ All data integrity checks passed'
    ELSE '❌ Found orphaned records'
  END;
"
```
