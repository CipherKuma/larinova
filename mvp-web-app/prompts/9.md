10. Create another prescription for different patient:
```bash
supabase db remote shell -c "
WITH another_prescription AS (
  INSERT INTO larinova_prescriptions (consultation_id, patient_id, doctor_id, doctor_notes)
  SELECT
    c.id,
    c.patient_id,
    c.doctor_id,
    'Rest and hydration recommended. Avoid strenuous activity.'
  FROM larinova_consultations c
  WHERE c.status = 'completed'
  AND NOT EXISTS (
    SELECT 1 FROM larinova_prescriptions pr WHERE pr.consultation_id = c.id
  )
  ORDER BY c.created_at DESC
  LIMIT 1
  RETURNING id, prescription_code
)
INSERT INTO larinova_prescription_items (prescription_id, medicine_id, medicine_name, dosage, frequency, duration, instructions)
SELECT
  ap.id,
  m.id,
  m.name,
  '10mg',
  'Once daily',
  '14 days',
  'Take before bed'
FROM another_prescription ap
CROSS JOIN larinova_medicines m
WHERE m.name = 'Cetirizine'
RETURNING medicine_name, dosage;
"
```

11. Get prescription statistics:
```bash
supabase db remote shell -c "
SELECT
  COUNT(DISTINCT p.id) as total_prescriptions,
  COUNT(pi.id) as total_medicine_items,
  AVG(item_count) as avg_medicines_per_prescription
FROM larinova_prescriptions p
LEFT JOIN larinova_prescription_items pi ON p.id = pi.prescription_id
LEFT JOIN (
  SELECT prescription_id, COUNT(*) as item_count
  FROM larinova_prescription_items
  GROUP BY prescription_id
) counts ON p.id = counts.prescription_id;
"
```
