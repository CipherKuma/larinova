8. Verify consultations and transcripts:
```bash
supabase db remote shell -c "
SELECT
  c.consultation_code,
  p.full_name as patient,
  c.status,
  c.duration_minutes,
  COUNT(t.id) as transcript_count
FROM larinova_consultations c
JOIN larinova_patients p ON c.patient_id = p.id
LEFT JOIN larinova_transcripts t ON c.id = t.consultation_id
GROUP BY c.consultation_code, p.full_name, c.status, c.duration_minutes
ORDER BY c.created_at DESC;
"
```

9. Check consultation statistics:
```bash
supabase db remote shell -c "
SELECT
  COUNT(*) as total_consultations,
  COUNT(CASE WHEN status = 'completed' THEN 1 END) as completed,
  COUNT(CASE WHEN status = 'in_progress' THEN 1 END) as in_progress,
  AVG(CASE WHEN duration_minutes IS NOT NULL THEN duration_minutes END)::numeric(10,2) as avg_duration
FROM larinova_consultations;
"
```
